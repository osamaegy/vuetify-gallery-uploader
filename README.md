# Vuetify Gallery Uploader

## Project setup
```
npm i vuetify-gallery-uploader
```
## Usage
```
import VuetifyGalleryUploader from "vuetify-gallery-uploader";

Vue.use(VuetifyGalleryUploader);
```

## Example
```
<gallery-uploader @submit="submit" />

methods:{
       submit(data){
            // List of all images
            console.log(data)
        }
    }

```
